import tkinter as tk
from tkinter import messagebox, ttk
import requests
from datetime import datetime

class WeatherApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Weather Forecast App")
        self.root.geometry("800x400")
        self.root.resizable(False, False)
        self.root.configure(bg="#F994DC")
        
        # API configuration
        self.api_key = "YOUR_API_KEY_HERE"
        self.base_url = "http://api.openweathermap.org/data/2.5/weather?"
        
        # Create GUI
        self.create_widgets()
    
    def create_widgets(self):
        # Main frame
        main_frame = ttk.Frame(self.root, padding="20")
        main_frame.pack(fill=tk.BOTH, expand=True)
        
        # City input
        ttk.Label(main_frame, text="City Name:").grid(row=0, column=0, padx=5, pady=5)
        self.city_entry = ttk.Entry(main_frame, width=30)
        self.city_entry.grid(row=0, column=1, padx=5, pady=5)
        
        # Country input (optional)
        ttk.Label(main_frame, text="Country Code:").grid(row=1, column=0, padx=5, pady=5)
        self.country_entry = ttk.Entry(main_frame, width=10)
        self.country_entry.grid(row=1, column=1, padx=5, pady=5, sticky="w")
        
        # Get Weather button
        self.get_btn = ttk.Button(main_frame, text="Get Weather", command=self.get_weather)
        self.get_btn.grid(row=2, column=0, columnspan=2, pady=15)
        
        # Results display
        self.result_text = tk.StringVar()
        self.result_text.set("Enter a city to get weather information")
        
        self.result_label = ttk.Label(main_frame, 
                                     textvariable=self.result_text,
                                     font=('Arial', 12),
                                     wraplength=400,
                                     justify="center")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=20)
    
    def get_weather(self):
        city = self.city_entry.get().strip()
        country = self.country_entry.get().strip()
        
        if not city:
            messagebox.showerror("Error", "Please enter a city name")
            return
        
        try:
            # Build query
            query = f"{self.base_url}q={city}"
            if country:
                query += f",{country}"
            query += f"&appid={self.api_key}&units=metric"
            
            # Fetch data
            response = requests.get(query)
            data = response.json()
            
            if data["cod"] != 200:
                messagebox.showerror("Error", data["message"])
                return
            
            # Extract data
            temp = data["main"]["temp"]
            feels_like = data["main"]["feels_like"]
            humidity = data["main"]["humidity"]
            wind_speed = data["wind"]["speed"]
            description = data["weather"][0]["description"].title()
            city_name = data["name"]
            country_code = data["sys"]["country"]
            
            # Update result
            result = f"üìç {city_name}, {country_code}\\n"
            result += f"üå°Ô∏è Temperature: {temp}¬∞C\\n"
            result += f"ü§î Feels like: {feels_like}¬∞C\\n"
            result += f"üíß Humidity: {humidity}%\\n"
            result += f"üí® Wind Speed: {wind_speed} m/s\\n"
            result += f"‚òÅÔ∏è Condition: {description}"
            
            self.result_text.set(result)
            
        except requests.exceptions.RequestException:
            messagebox.showerror("Error", "Unable to connect to weather service")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")

if __name__ == "__main__":
    root = tk.Tk()
    app = WeatherApp(root)
    root.mainloop()
